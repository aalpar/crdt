**Goal:** Implement a delta-state RGA (Replicated Growable Array) composing `DotFun[Node[E]]`.

**Architecture:** `Causal[*DotFun[Node[E]]]` where each sequence element is a dot. `Node[E]` stores value, predecessor reference, and tombstone flag. Tombstoned nodes persist as tree anchors. Linearization reconstructs order by walking the `After` tree with sibling tiebreak (seq DESC, ID DESC).

**Tech Stack:** Go 1.25, `dotcontext` package, `quicktest` for tests, `crdttest.Harness` for semilattice properties.

```go
package rga

import (
	"github.com/aalpar/crdt/dotcontext"
)

// Node is a single element in the RGA sequence.
// It implements dotcontext.Lattice: join is monotonic on Deleted
// (once tombstoned, stays tombstoned).
//
// Under normal operation, JoinDotFun only calls Join when both
// sides share the same dot, meaning Value and After are identical.
// The Deleted flag is the only field that can differ across replicas.
type Node[E comparable] struct {
	Value   E
	After   dotcontext.Dot // zero Dot = head of list
	Deleted bool
}

// Join satisfies dotcontext.Lattice. Delete wins: if either
// side is tombstoned, the result is tombstoned.
func (n Node[E]) Join(other Node[E]) Node[E] {
	if other.Deleted {
		return Node[E]{Value: n.Value, After: n.After, Deleted: true}
	}
	return n
}

// Element is a visible (non-tombstoned) entry in the sequence.
type Element[E comparable] struct {
	ID    dotcontext.Dot
	Value E
}
```

```go
// RGA is a replicated growable array. Elements are immutable once
// inserted. Concurrent inserts at the same position are ordered
// deterministically by dot comparison (seq DESC, ID DESC).
//
// Internally it is a DotFun[Node[E]] paired with a causal context.
// Deleted elements persist as tombstones to anchor ordering.
type RGA[E comparable] struct {
	id    dotcontext.ReplicaID
	state dotcontext.Causal[*dotcontext.DotFun[Node[E]]]
}

// New creates an empty RGA for the given replica.
func New[E comparable](replicaID dotcontext.ReplicaID) *RGA[E] {
	return &RGA[E]{
		id: replicaID,
		state: dotcontext.Causal[*dotcontext.DotFun[Node[E]]]{
			Store:   dotcontext.NewDotFun[Node[E]](),
			Context: dotcontext.New(),
		},
	}
}

// State returns the RGA's internal Causal state for serialization.
func (r *RGA[E]) State() dotcontext.Causal[*dotcontext.DotFun[Node[E]]] {
	return r.state
}

// FromCausal constructs an RGA from a decoded Causal value.
func FromCausal[E comparable](state dotcontext.Causal[*dotcontext.DotFun[Node[E]]]) *RGA[E] {
	return &RGA[E]{state: state}
}

// Merge incorporates a delta or full state from another RGA.
func (r *RGA[E]) Merge(other *RGA[E]) {
	dotcontext.MergeDotFun(&r.state, other.state)
}
```

```go
// InsertAfter inserts value after the element identified by after.
// Use zero Dot to insert at the head of the list.
// Returns a delta for replication.
func (r *RGA[E]) InsertAfter(after dotcontext.Dot, value E) *RGA[E] {
	d := r.state.Context.Next(r.id)
	node := Node[E]{Value: value, After: after}

	r.state.Store.Set(d, node)

	deltaStore := dotcontext.NewDotFun[Node[E]]()
	deltaStore.Set(d, node)

	deltaCtx := dotcontext.New()
	deltaCtx.Add(d)

	return &RGA[E]{
		state: dotcontext.Causal[*dotcontext.DotFun[Node[E]]]{
			Store:   deltaStore,
			Context: deltaCtx,
		},
	}
}

// Delete marks the element at the given dot as removed.
// Returns a delta for replication. No-op if the dot is missing
// or already deleted.
func (r *RGA[E]) Delete(at dotcontext.Dot) *RGA[E] {
	node, ok := r.state.Store.Get(at)
	if !ok || node.Deleted {
		return &RGA[E]{
			state: dotcontext.Causal[*dotcontext.DotFun[Node[E]]]{
				Store:   dotcontext.NewDotFun[Node[E]](),
				Context: dotcontext.New(),
			},
		}
	}

	tombstoned := Node[E]{Value: node.Value, After: node.After, Deleted: true}
	r.state.Store.Set(at, tombstoned)

	deltaStore := dotcontext.NewDotFun[Node[E]]()
	deltaStore.Set(at, tombstoned)

	deltaCtx := dotcontext.New()
	deltaCtx.Add(at)

	return &RGA[E]{
		state: dotcontext.Causal[*dotcontext.DotFun[Node[E]]]{
			Store:   deltaStore,
			Context: deltaCtx,
		},
	}
}

// linearize walks the After-tree and returns all nodes in sequence order.
// Siblings (concurrent inserts after the same parent) are ordered by
// dot: Seq descending, then ID descending.
func (r *RGA[E]) linearize() []dotcontext.Dot {
	// Build children map: parent → sorted children.
	children := make(map[dotcontext.Dot][]dotcontext.Dot)
	r.state.Store.Range(func(d dotcontext.Dot, n Node[E]) bool {
		children[n.After] = append(children[n.After], d)
		return true
	})

	// Sort each child list: higher seq first, higher ID first (leftmost).
	for _, kids := range children {
		sort.Slice(kids, func(i, j int) bool {
			if kids[i].Seq != kids[j].Seq {
				return kids[i].Seq > kids[j].Seq
			}
			return kids[i].ID > kids[j].ID
		})
	}

	// DFS from head (zero Dot).
	var order []dotcontext.Dot
	var walk func(parent dotcontext.Dot)
	walk = func(parent dotcontext.Dot) {
		for _, child := range children[parent] {
			order = append(order, child)
			walk(child)
		}
	}
	walk(dotcontext.Dot{})

	return order
}

// Elements returns the visible (non-tombstoned) elements in sequence order.
func (r *RGA[E]) Elements() []Element[E] {
	order := r.linearize()
	var elems []Element[E]
	for _, d := range order {
		n, _ := r.state.Store.Get(d)
		if !n.Deleted {
			elems = append(elems, Element[E]{ID: d, Value: n.Value})
		}
	}
	return elems
}

// Len returns the number of visible (non-tombstoned) elements.
func (r *RGA[E]) Len() int {
	n := 0
	r.state.Store.Range(func(_ dotcontext.Dot, node Node[E]) bool {
		if !node.Deleted {
			n++
		}
		return true
	})
	return n
}
```

```go
// At returns the element at the given index (0-based) in the visible
// sequence. Returns false if index is out of bounds.
func (r *RGA[E]) At(index int) (Element[E], bool) {
	if index < 0 {
		return Element[E]{}, false
	}
	order := r.linearize()
	i := 0
	for _, d := range order {
		n, _ := r.state.Store.Get(d)
		if !n.Deleted {
			if i == index {
				return Element[E]{ID: d, Value: n.Value}, true
			}
			i++
		}
	}
	return Element[E]{}, false
}
```

```go
// InsertAt inserts value at the given index in the visible sequence.
// Index 0 inserts before the first visible element (at head).
// Index equal to Len() appends at the end.
// Returns a delta for replication, or nil delta if index is out of bounds.
func (r *RGA[E]) InsertAt(index int, value E) *RGA[E] {
	if index < 0 {
		return &RGA[E]{
			state: dotcontext.Causal[*dotcontext.DotFun[Node[E]]]{
				Store:   dotcontext.NewDotFun[Node[E]](),
				Context: dotcontext.New(),
			},
		}
	}

	if index == 0 {
		return r.InsertAfter(dotcontext.Dot{}, value)
	}

	// Find the dot at index-1 in the visible sequence.
	e, ok := r.At(index - 1)
	if !ok {
		return &RGA[E]{
			state: dotcontext.Causal[*dotcontext.DotFun[Node[E]]]{
				Store:   dotcontext.NewDotFun[Node[E]](),
				Context: dotcontext.New(),
			},
		}
	}
	return r.InsertAfter(e.ID, value)
}

// DeleteAt deletes the element at the given index in the visible sequence.
// Returns a delta for replication.
func (r *RGA[E]) DeleteAt(index int) *RGA[E] {
	e, ok := r.At(index)
	if !ok {
		return &RGA[E]{
			state: dotcontext.Causal[*dotcontext.DotFun[Node[E]]]{
				Store:   dotcontext.NewDotFun[Node[E]](),
				Context: dotcontext.New(),
			},
		}
	}
	return r.Delete(e.ID)
}
```

```go
func TestConcurrentInserts(t *testing.T) {
	c := qt.New(t)

	c.Run("ConcurrentAtHead", func(c *qt.C) {

		b := New[string]("b")

		db := b.InsertAfter(dotcontext.Dot{}, "from-b")

		a.Merge(db)
		b.Merge(da)

		ae := a.Elements()
		be := b.Elements()
		c.Assert(len(ae), qt.Equals, 2)
		c.Assert(len(be), qt.Equals, 2)
		// Both converge to the same order.
		c.Assert(ae[0].Value, qt.Equals, be[0].Value)
		c.Assert(ae[1].Value, qt.Equals, be[1].Value)
	})

	c.Run("ConcurrentAfterSameElement", func(c *qt.C) {
		a := New[string]("a")
		b := New[string]("b")

		// Both insert "root" at head.
		da0 := a.InsertAfter(dotcontext.Dot{}, "root")
		b.Merge(da0)

		rootDot := a.Elements()[0].ID

		// Both insert after "root" concurrently.
		da := a.InsertAfter(rootDot, "from-a")
		db := b.InsertAfter(rootDot, "from-b")

		a.Merge(db)
		b.Merge(da)

		ae := a.Elements()
		be := b.Elements()
		c.Assert(len(ae), qt.Equals, 3)
		// Converge.
		for i := range ae {
			c.Assert(ae[i].Value, qt.Equals, be[i].Value)
		}
		// "root" is first.
		c.Assert(ae[0].Value, qt.Equals, "root")
	})

	c.Run("InsertAfterDeletedNode", func(c *qt.C) {
		a := New[string]("a")
		b := New[string]("b")

		// a inserts "x", replicates to b.
		da0 := a.InsertAfter(dotcontext.Dot{}, "x")
		b.Merge(da0)

		b.Merge(ddel)

		// Both converge: "x" is deleted, "y" survives anchored to x's position.
		ae := a.Elements()
		be := b.Elements()
		c.Assert(len(ae), qt.Equals, 1)
		c.Assert(ae[0].Value, qt.Equals, "y")
		c.Assert(len(be), qt.Equals, 1)
		c.Assert(be[0].Value, qt.Equals, "y")
```

```go
// Package rga implements a replicated growable array, a delta-state
// CRDT for ordered sequences where elements are immutable once
// inserted.
//
// The implementation composes DotFun[Node[E]] from the dotcontext
// package. Each element is identified by a unique dot and stores a
// value plus a reference to its predecessor. Concurrent inserts at
// the same position are ordered deterministically by dot comparison
// (higher sequence number first, then higher replica ID).
//
// Deletion tombstones the element rather than removing it, preserving
// the tree structure that anchors ordering. Tombstoned elements are
// invisible in query results but persist in the store.
//
// Mutators return deltas suitable for replication. Merge incorporates
// a delta or full state from another replica.
```
